<script>

    const moods = {

        "-4": {e:"нµ≥", n:"еЕіе•Л", c:"var(--m-excite)"}, "-3": {e:"нµ∞", n:"ењЂдєР", c:"var(--m-happy)"},

        "-2": {e:"нЄМ", n:"жї°иґ≥", c:"var(--m-content)"}, "-1": {e:"нЄК", n:"жМЇе•љ", c:"var(--m-good)"},

        "0":  {e:"нЄР", n:"еє≥еТМ", c:"var(--m-peace)"},  "1":  {e:"ніР", n:"еЖНењНењН", c:"var(--m-bear)"},

        "2":  {e:"нЄ∞", n:"зД¶иЩС", c:"var(--m-anxiety)"}, "3":  {e:"ніђ", n:"зФЯж∞Ф", c:"var(--m-angry)"},

        "4":  {e:"нЄЂ", n:"зЕОзЖђ", c:"var(--m-struggle)"}, "5":  {e:"н≤А", n:"зЧЫиЛ¶", c:"var(--m-pain)"}

    };

 

    function handleDateChange(val) {

        if(!val) return;

        const d = new Date(val);

        const weeks = ["еС®жЧ•","еС®дЄА","еС®дЇМ","еС®дЄЙ","еС®еЫЫ","еС®дЇФ","еС®еЕ≠"];

        document.getElementById('dateLabel').innerText = `${d.getFullYear()}еєі${d.getMonth()+1}жЬИ${d.getDate()}жЧ• ${weeks[d.getDay()]}`;

        document.getElementById('wallMonthTitle').innerText = `${d.getMonth()+1}жЬИ зП≠зП≠жФґзЇ≥еҐЩ`;

        renderMoodWall();

    }

 

    function syncTime(v) {

        document.getElementById('inH').value = Math.floor(v / 60);

        document.getElementById('inM').value = v % 60;

    }

 

    function updateMoodUI() {

        const v = document.getElementById('moodS').value;

        const m = moods[v];

        const box = document.getElementById('moodBox');

        box.style.backgroundColor = m.c;

        document.getElementById('moodIcon').innerText = m.e;

        document.getElementById('moodName').innerText = m.n;

        box.style.color = (v >= 3 || v <= -3) ? "#fff" : "var(--main-ink)";

    }

 

    function toggleStats(show) {

        document.getElementById('statsPage').style.display = show ? 'flex' : 'none';

        if(show) renderStatsList();

    }

 

    function saveData() {

        const txt = document.getElementById('txtInp').value;

        if(!txt) return;

        const moodVal = document.getElementById('moodS').value;

        const selDate = new Date(document.getElementById('datePicker').value);

        const entry = {

            id: Date.now(),

            dateLabel: document.getElementById('dateLabel').innerText,

            ym: `${selDate.getFullYear()}-${selDate.getMonth()+1}`,

            d: selDate.getDate(),

            moodC: moods[moodVal].c, moodE: moods[moodVal].e, moodN: moods[moodVal].n,

            content: txt, time: `${document.getElementById('inH').value}h ${document.getElementById('inM').value}m`

        };

        let storage = JSON.parse(localStorage.getItem('banban_v17_final') || '[]');

        const idx = storage.findIndex(r => r.ym === entry.ym && r.d === entry.d);

        if(idx > -1) storage[idx] = entry; else storage.unshift(entry);

        localStorage.setItem('banban_v17_final', JSON.stringify(storage));

        

        document.getElementById('txtInp').value = "";

        renderMoodWall();

        alert("жФґзЇ≥жИРеКЯпЉБ");

    }

 

    function renderMoodWall() {

        const grid = document.getElementById('wallGrid'); grid.innerHTML = '';

        const storage = JSON.parse(localStorage.getItem('banban_v17_final') || '[]');

        const selDate = new Date(document.getElementById('datePicker').value);

        const y = selDate.getFullYear(), m = selDate.getMonth();

        const days = new Date(y, m + 1, 0).getDate();

        for (let i = 1; i <= days; i++) {

            const r = storage.find(x => x.ym === `${y}-${m+1}` && x.d === i);

            const el = document.createElement('div');

            el.className = 'wall-day' + (r ? ' active' : '');

            if(r) { el.style.background = r.moodC; el.innerHTML = `<span>${r.moodE}</span>`; }

            else { el.innerText = i; }

            grid.appendChild(el);

        }

    }

 

    function renderStatsList() {

        const data = JSON.parse(localStorage.getItem('banban_v17_final') || '[]');

        document.getElementById('statsList').innerHTML = data.length ? data.map(item => `

            <div class="list-item">

                <div class="axis"><div class="dot"></div><div class="line"></div></div>

                <div class="log-card">

                    <div class="mood-strip" style="background:${item.moodC}"></div>

                    <div class="log-body">

                        <div class="log-date">${item.dateLabel}</div>

                        <div class="log-txt">${item.content}</div>

                    </div>

                </div>

            </div>

        `).join('') : '<p style="text-align:center; color:#ccc; margin-top:50px;">ињШж≤°жЬЙиЃ∞ељХеУ¶</p>';

    }

 

    window.onload = () => {

        const today = new Date().toISOString().split('T')[0];

        document.getElementById('datePicker').value = today;

        handleDateChange(today);

        updateMoodUI();

    };
